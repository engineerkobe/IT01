--rfid:卡片的號碼
--is_borrow:是否被借出
--borrower:借書人帳號
--log:借出的日期與時間，預設時間1900-00-00，1900-00-00代表沒有借出
--isbn:rfid_book的isbn(外鍵)對應表isbn_book的isbn(主鍵)
--
CREATE DATABASE Demo2 CHARACTER SET utf8 COLLATE utf8_general_ci;

CREATE TABLE rfid_book(
	rfid VARCHAR(50) NOT NULL PRIMARY KEY,
	is_borrow VARCHAR(1) NOT NULL,
	borrower VARCHAR(50) NOT NULL,
	log TIME NOT NULL,
	isbn VARCHAR(50) NOT NULL,

	FOREIGN KEY (isbn)
	REFERENCES isbn_book (isbn)
);



CREATE TABLE isbn_book(
	isbn VARCHAR(50) NOT NULL PRIMARY KEY,
	name VARCHAR(255) NOT NULL ,
	content TEXT NOT NULL,
	count INT NOT NULL
);


SELECT count(*) FROM isbn_book,rfid_book 
WHERE isbn_book.isbn = rfid_book.isbn;
